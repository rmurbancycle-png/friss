npm install express ws node friss.js const express = require('express'); const http = require('http'); const WebSocket = require('ws'); const app = express(); app.use(express.json()); // Simulación de Base de Datos en memoria para el Mercado del Val let pedidos = []; let comerciosActivos = { "mesetarios": { nombre: "Mesetarios (Embutidos)", ws: null }, "el-bombon": { nombre: "El Bombón (Pastelería)", ws: null }, "veggs": { nombre: "Vegg's (Plant-based)", ws: null } }; let ridersActivos = []; // --- SERVIDOR HTTP --- // 1. Endpoint para crear un pedido combinado app.post('/api/pedido', (req, res) => { const { cliente, direccion, productos } = req.body; const nuevoPedido = { id: pedidos.length + 1, cliente, direccion, productos: productos.map(p => ({ ...p, estado: 'Pendiente' })), // Cada producto empieza pendiente estadoGeneral: 'Preparando en el Mercado', // <-- CORREGIDO: Sin espacios fecha: new Date() }; pedidos.push(nuevoPedido); // Notificar a los comercios implicados productos.forEach(p => { const comercio = comerciosActivos[p.comercio]; if (comercio && comercio.ws && comercio.ws.readyState === WebSocket.OPEN) { comercio.ws.send(JSON.stringify({ evento: 'NUEVO_PEDIDO_PRODUCTO', pedidoId: nuevoPedido.id, producto: p.item, direccion: nuevoPedido.direccion })); } }); res.status(201).json({ mensaje: "Pedido procesado y notificado al Mercado del Val", pedidoId: nuevoPedido.id }); }); // 2. Historial de pedidos app.get('/api/pedidos', (req, res) => { res.json(pedidos); }); // --- SERVIDOR WEBSOCKET (Tiempo Real) --- const server = http.createServer(app); const wss = new WebSocket.Server({ server }); wss.on('connection', (ws, req) => { const urlParams = new URLSearchParams(req.url.split('?')[1]); const rol = urlParams.get('rol'); if (comerciosActivos[rol]) { comerciosActivos[rol].ws = ws; console.log(`🏪 Comercio conectado al Friss Hub: ${comerciosActivos[rol].nombre}`); } else if (rol === 'rider') { ridersActivos.push(ws); console.log(`🛵 Repartidor conectado en Valladolid listo para reparto.`); } // LÓGICA DE MENSAJES ws.on('message', (message) => { try { const data = JSON.parse(message); console.log(`[Evento recibido de ${rol}]:`, data); // Acción: El comercio avisa que terminó su producto if (data.evento === 'PRODUCTO_LISTO') { let pedido = pedidos.find(p => p.id === data.pedidoId); if (pedido) { let prod = pedido.productos.find(p => p.comercio === rol); if (prod) prod.estado = 'Listo'; // Comprobar si TODOS los productos del pedido están listos let todoListo = pedido.productos.every(p => p.estado === 'Listo'); if (todoListo) { pedido.estadoGeneral = 'Listo para recogida'; // Avisar inmediatamente a los riders de Valladolid ridersActivos.forEach(rider => { if (rider.readyState === WebSocket.OPEN) { rider.send(JSON.stringify({ evento: 'NUEVO_VIAJE_DISPONIBLE', pedidoId: pedido.id, recogida: 'Mercado del Val (Hub)', destino: pedido.direccion })); } }); } } } } catch (err) { console.error("Error al procesar mensaje JSON:", err); } }); ws.on('close', () => { if (comerciosActivos[rol] && comerciosActivos[rol].ws === ws) { comerciosActivos[rol].ws = null; } ridersActivos = ridersActivos.filter(r => r !== ws); console.log(`❌ Conexión cerrada para el rol: ${rol}`); }); }); const PORT = 3000; server.listen(PORT, () => { console.log(`🚀 Friss Engine corriendo sin fallos en http://localhost:${PORT}`); console.log(`📍 Hub Centralizado: Mercado del Val, Valladolid.`); });
